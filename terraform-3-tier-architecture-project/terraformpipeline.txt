
pipeline {
	agent any
    environment {
        AWS_ACCESS_KEY_ID     = credentials('accesskey')
        AWS_SECRET_ACCESS_KEY = credentials('secretkey')
	}
	stage {
		stage('checkout') {
			steps {
				git 'https://github.com/devops/terraform-3-tier-arch.git'
				}
		}
		stage('init') {
			steps {
				sh 'terraform init'
			}
		}
		stage('valid') {
			steps {
				sh 'terraform validate'
			}
		}
		stage('plan') {
			steps {
				sh 'terraform plan'
			}
		}
		stage('action') {
			steps {
				sh 'terraform $action --auto-approve'
			}
		}
	}
}
